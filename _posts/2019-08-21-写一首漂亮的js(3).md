## 适当的使用三元表达式子

```javascript
// 当只有一行执行语句的时候，使用三元表达式
// bad
if (isEdit) {
  update()
} else {
  reset()
}

// good
isEdit ? update() : reset()

// 当有多行执行语句的时候
// good
if (isEdit) {
  one()
  two()
  update()
} else {
  reset()
}
```

## 提前返回

```javascript
// 当只有一行执行语句的时候
// emmm... ok, not bad
if (isOk) {
  one()
}

// 且表达式
// good
isOk && one()

// 当有多行执行语句的时候
// bad
if (isOk) {
  one()
  tow()

}

// good
if (!isOk) {
  return
}

one()
```
